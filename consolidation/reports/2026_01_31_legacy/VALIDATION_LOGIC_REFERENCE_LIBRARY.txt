BLOCK PARSING FROM ADDRESS:
───────────────────────────

Logic:
  1. If address contains " & " → Preserve as intersection
  2. Otherwise:
     - Extract street number
     - Extract street name
     - Create "X Block of STREET NAME" format
     - Block number = (street number // 100) * 100

PYTHON IMPLEMENTATION:
```python
def parse_block(addr):
    if pd.isna(addr):
        return None
    
    # Preserve intersections
    if "&" in addr:
        return addr
    
    # Parse standard address
    try:
        # Remove city/state/zip
        clean = addr.replace(", Hackensack, NJ, 07601", "")
        parts = clean.split(" ", 1)
        
        # Extract number and street
        num = int(parts[0])
        street = parts[1].split(",", 1)[0].strip()
        
        # Calculate block number
        block_num = (num // 100) * 100
        
        return f"{block_num} Block of {street}"
    except:
        return "Check CAD Location Data"

df["Block"] = df["FullAddress2"].apply(parse_block)
```

TIME RESPONSE CALCULATION:
──────────────────────────

Formula: Time Dispatched - Time of Call

Rules:
  - Handle negative results by swapping timestamps
  - Cap outliers at 480 minutes (8 hours)
  - Calculate in minutes
  - Format as "M Mins S Secs"

PYTHON IMPLEMENTATION:
```python
def calculate_time_response(time_call, time_dispatched):
    if pd.isna(time_call) or pd.isna(time_dispatched):
        return None
    
    # Calculate difference
    diff = (time_dispatched - time_call).total_seconds() / 60
    
    # Handle negatives (swap timestamps)
    if diff < 0:
        diff = abs(diff)
    
    # Cap at 480 minutes (8 hours)
    if diff > 480:
        diff = 480
    
    return diff

df["Time_Response_Minutes"] = df.apply(
    lambda row: calculate_time_response(
        row["Time of Call"], 
        row["Time Dispatched"]

)
```

```python
def format_response_time(mins):
    if pd.isna(mins):

    total_sec = round(mins * 60)
```

CAD NOTES PARSING:
──────────────────

Extract from CADNotes:
  1. CAD_Username:  Username from header (contains underscore)
  2. CAD_Timestamp: Timestamp from header (MM/DD/YYYY HH:MM:SS format)
  3. CAD_Notes_Cleaned: Note text with metadata removed

PYTHON IMPLEMENTATION:
```python
import re

def parse_cad_notes(raw):
    if pd.isna(raw): 
        return pd.Series([None, None, None])
    
    lines = [l for l in raw.splitlines() if l.strip()]
    header = lines[0] if lines else ""
    tokens = header.split()
    
    # Extract username (contains underscore)
    user = next((t for t in tokens if "_" in t), None)
    
    # Extract timestamp
    dt_raw = ""
    for i, t in enumerate(tokens):
        if "/" in t and i + 1 < len(tokens):
            dt_raw = t + " " + tokens[i + 1]
            break
    
    # Parse timestamp
    try:
        dt = pd.to_datetime(dt_raw)
        ts = dt.strftime("%m/%d/%Y %H:%M:%S")
    except:
        ts = None
    
    # Clean note text
    cleaned = raw.replace(header, "").replace(dt_raw, "")
    if user:
        cleaned = cleaned.replace(user, "")
    
    # Remove remaining date patterns
    cleaned = " ".join([
        w for w in cleaned.split() 
        if not re.match(r"\d{1,2}/\d{1,2}/", w)
    ])
    
    cleaned = cleaned.strip().title() or None
    
    return pd.Series([user, ts, cleaned])

df[["CAD_Username", "CAD_Timestamp", "CAD_Notes_Cleaned"]] = (
    df["CADNotes"].apply(parse_cad_notes)
)
```

ADDRESS FORMAT VALIDATION:
──────────────────────────

Required format: {Number} {Street Name} {Street Type}, Hackensack, NJ, 07601

Validation rules:
  1. Must contain street number (digits at start)
  2. Must contain street name
  3. Must contain "Hackensack, NJ"
  4. Must NOT contain "Jersey City" or other non-Hackensack locations
  5. Intersection format: {Street 1} & {Street 2}

PYTHON IMPLEMENTATION:
```python
import re

def validate_address(addr):
    """
    Validate address format against ArcGIS standards.
    Returns: (is_valid: bool, error_message: str)
    """
    if pd.isna(addr):
        return False, "Address is null"
    
    addr_clean = str(addr).strip()
    
    # Check for required city
    if "Hackensack, NJ" not in addr_clean:
        if "Jersey City" in addr_clean:
            return False, "Wrong city (Jersey City)"
        return False, "Missing 'Hackensack, NJ'"
    
    # Check for intersection format
    if " & " in addr_clean:
        # Intersection should have two street names
        parts = addr_clean.split(" & ")
        if len(parts) != 2:
            return False, "Invalid intersection format"
        if not parts[0].strip() or not parts[1].strip():
            return False, "Missing street name in intersection"
        return True, None
    
    # Check for standard address format
    # Should start with number
    match = re.match(r'^(\d+)\s+(.+)', addr_clean)
    if not match:
        return False, "Missing street number"
    
    street_number = match.group(1)
    street_part = match.group(2)
    
    # Should have street name
    if not street_part or len(street_part) < 3:
        return False, "Missing or invalid street name"
    
    return True, None

# Apply validation
df[["Address_Valid", "Address_Error"]] = df["FullAddress2"].apply(
    lambda x: pd.Series(validate_address(x))
)

# Count errors
address_errors = (~df["Address_Valid"]).sum()
```

```python
def validate_street_name(addr, official_streets):
    """
    Validate street name against official list.
    Returns: (is_valid: bool, suggested_name: str)
    """
    if pd.isna(addr) or " & " in addr:
        return True, None  # Skip intersections
    
    # Extract street name
    match = re.match(r'^\d+\s+(.+?),', addr)
    if not match:
        return False, None
    
    street_full = match.group(1).strip()
    
    # Check exact match
    if street_full in official_streets:
        return True, None
    
    # Check fuzzy match (Levenshtein distance)
    from difflib import get_close_matches
    matches = get_close_matches(street_full, official_streets, n=1, cutoff=0.8)
    
    if matches:
        return False, matches[0]  # Suggest closest match
    
    return False, None  # No match found
```

RESPONSE TIME VALIDATION:
─────────────────────────

Rules:
  1. Must be non-negative (Time Dispatched >= Time of Call)
  2. Must be < 480 minutes (8 hours)
  3. If negative, flag as error (timestamps swapped)
  4. If > 480 minutes, flag as outlier

PYTHON IMPLEMENTATION:
```python
def validate_response_time(time_call, time_dispatched):
    """
    Validate response time calculation.
    Returns: (is_valid: bool, error_type: str, corrected_value: float)
    """
    if pd.isna(time_call) or pd.isna(time_dispatched):
        return False, "Missing timestamp", None
    
    # Calculate difference
    diff_minutes = (time_dispatched - time_call).total_seconds() / 60
    
    # Check for negative (timestamps swapped)
    if diff_minutes < 0:
        corrected = abs(diff_minutes)
        if corrected <= 480:
            return False, "Negative time (timestamps swapped)", corrected
        else:
            return False, "Invalid timestamps (difference > 8 hours)", None
    
    # Check for outlier
    if diff_minutes > 480:
        return False, "Outlier (> 8 hours)", 480  # Cap at 480
    
    return True, None, diff_minutes

# Apply validation
df[["Response_Valid", "Response_Error", "Response_Corrected"]] = (

            validate_response_time(
                row["Time of Call"],

        axis=1
```

```python
def validate_timestamp_sequence(row):
    """
    Validate timestamp sequence is logical.
    Returns: list of errors
    """
    errors = []
    
    time_call = row["Time of Call"]
    time_dispatched = row["Time Dispatched"]
    time_out = row["Time Out"]
    time_in = row["Time In"]
    
    # Check call <= dispatched
    if pd.notna(time_call) and pd.notna(time_dispatched):
        if time_call > time_dispatched:
            errors.append("Call time > Dispatch time")
    
    # Check dispatched <= out
    if pd.notna(time_dispatched) and pd.notna(time_out):
        if time_dispatched > time_out:
            errors.append("Dispatch time > Out time")
    
    # Check out <= in
    if pd.notna(time_out) and pd.notna(time_in):
        if time_out > time_in:
            errors.append("Out time > In time")
    
    return errors if errors else None

df["Timestamp_Errors"] = df.apply(validate_timestamp_sequence, axis=1)
```

```python
def calculate_null_rates(df):
    """Calculate null rates for all columns."""
    null_rates = {}
    
    for col in df.columns:
        null_count = df[col].isna().sum()
        null_rate = (null_count / len(df)) * 100
        null_rates[col] = {
            "null_count": null_count,
            "null_rate": null_rate
        }
    
    return pd.DataFrame(null_rates).T
```

```python
def check_completeness(df):
    """
    Check data completeness.
    Returns: completeness report
    """
    report = {
        "total_records": len(df),
        "fields_checked": {},
        "overall_score": 0
    }
    
    # Define expected fields and weights
    field_weights = {
        "ReportNumberNew": 10,
        "Incident": 8,
        "Incident_Date": 10,
        "Incident_Time": 8,
        "How Reported": 7,
        "Disposition": 7,
        "FullAddress2": 8,
        "Zone": 6,
        "Grid": 5,
        "Officer": 6,
        "Time of Call": 6,
        "Time Dispatched": 5,
        "Time Out": 4,
        "Time In": 4
    }
    
    total_weight = sum(field_weights.values())
    weighted_score = 0
    
    for field, weight in field_weights.items():
        if field in df.columns:
            null_count = df[field].isna().sum()
            populated_rate = ((len(df) - null_count) / len(df)) * 100
            field_score = populated_rate * weight / 100
            weighted_score += field_score
            
            report["fields_checked"][field] = {
                "exists": True,
                "null_count": int(null_count),
                "populated_rate": round(populated_rate, 2),
                "weight": weight,
                "weighted_score": round(field_score, 2)
            }
        else:
            report["fields_checked"][field] = {
                "exists": False,
                "null_count": len(df),
                "populated_rate": 0,
                "weight": weight,
                "weighted_score": 0
            }
    
    report["overall_score"] = round((weighted_score / total_weight) * 100, 2)
    
    return report

# Run completeness check
completeness_report = check_completeness(df)
print(f"Overall Completeness Score: {completeness_report['overall_score']}%")
```
