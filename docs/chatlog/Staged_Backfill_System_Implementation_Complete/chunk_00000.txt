 def calculate_sha256(file_path: Path) -> str:
     """Calculate SHA256 hash of file for integrity verification.""" sha256_hash = hashlib.sha256()
     with open(file_path, "rb") as f:
         for byte_block in iter(lambda: f.read(4096), b""):
             sha256_hash.update(byte_block)
     return sha256_hash.hexdigest()


 def load_baseline_file(baseline_path: Path) -> pd.DataFrame:
     """Load baseline polished Excel file into pandas DataFrame.""" print(f"\n[1/6] Loading baseline file: {baseline_path.name}")

 def extract_unique_addresses(df: pd.DataFrame) -> pd.DataFrame:
     """Extract unique FullAddress values for geocoding.""" print("\n[2/6] Extracting unique addresses for geocoding")

     # Check if FullAddress column exists
     if "FullAddress" not in df.columns:
         # Try common alternatives
         address_cols = [col for col in df.columns if "address" in col.lower()]
         if address_cols:
             print(f"      [WARNING] 'FullAddress' not found, using: {address_cols[0]}")
             df = df.rename(columns={address_cols[0]: "FullAddress"})
         else:
             raise ValueError("No address column found in baseline file")

     # Extract unique addresses (exclude nulls)
     unique_addresses = df[["FullAddress"]].drop_duplicates()
     unique_addresses = unique_addresses[unique_addresses["FullAddress"].notna()]

     print(f"      Total records: {len(df):,}")
     print(f"      Unique addresses: {len(unique_addresses):,}")
     print(f"      Compression ratio: {(1 - len(unique_addresses) / len(df)) * 100:.1f}%")

     return unique_addresses


 def geocode_addresses(unique_addresses: pd.DataFrame) -> pd.DataFrame:
     """
     Batch geocode unique addresses using Esri World Geocoder. IMPORTANT: This requires ArcGIS Pro license with credits. Pauses before execution to allow user to verify license.
     """ print("\n[3/6] Geocoding unique addresses (ONE-TIME OPERATION)")
     print(f"      Geocoder: {GEOCODER_URL}")
     print(f"      Addresses to geocode: {len(unique_addresses):,}")
     print("\n      [PAUSE] This operation requires ArcGIS Pro license with credits.") print("              Press Ctrl+C within 10 seconds to abort if license not available.") try:
         time.sleep(10)
     except KeyboardInterrupt:
         print("\n[ABORT] User cancelled geocoding operation.") sys.exit(0)

     print("\n      Starting geocoding...")
     start_time = time.time()

     # Create in-memory table from DataFrame
     temp_table = r"memory\unique_addresses"
     temp_feature_class = r"memory\geocoded_cache"

     try:
         # Convert DataFrame to in-memory table
         array = []
         for idx, row in unique_addresses.iterrows():
             array.append((row["FullAddress"],))

         # Create table
         arcpy.management.CreateTable(r"memory", "unique_addresses")
         arcpy.management.AddField(temp_table, "FullAddress", "TEXT", field_length=255)

         # Insert addresses
         with arcpy.da.InsertCursor(temp_table, ["FullAddress"]) as cursor:
             for addr in array:
                 cursor.insertRow(addr)

         # Geocode addresses
         arcpy.geocoding.GeocodeAddresses(
             in_table=temp_table,
             address_locator=GEOCODER_URL,
             in_address_fields="'Single Line Input' FullAddress",
             out_feature_class=temp_feature_class,
             out_relationship_type="ONE_TO_ONE"
         )

         # Extract X/Y coordinates from geocoded results
         geocoded_results = []
         with arcpy.da.SearchCursor(
             temp_feature_class,
             ["FullAddress", "SHAPE@XY", "Status"]
         ) as cursor:
             for row in cursor:
                 address, coords, status = row
                 if coords and status == "M":  # Matched
                     x, y = coords
                     geocoded_results.append({
                         "FullAddress": address,
                         "X_Coord": x,
                         "Y_Coord": y,
                         "GeocodeStatus": status
                     })
                 else:
                     geocoded_results.append({
                         "FullAddress": address,
                         "X_Coord": None,
                         "Y_Coord": None,
                         "GeocodeStatus": status or "U"  # Unmatched
                     })

         # Convert to DataFrame
         cache_df = pd.DataFrame(geocoded_results)

         elapsed = time.time() - start_time
         print(f"      Geocoding complete in {elapsed:.1f} seconds")
         print(f"      Rate: {len(unique_addresses) / elapsed:.1f} addresses/sec")

         return cache_df

     except arcpy.ExecuteError as e:
         print(f"\n[ERROR] ArcPy geocoding failed: {e}")
         print(arcpy.GetMessages())
         sys.exit(1)
     except Exception as e:
         print(f"\n[ERROR] Unexpected error during geocoding: {e}")
         sys.exit(1)
     finally:
         # Cleanup temporary datasets
         for temp in [temp_table, temp_feature_class]:
             if arcpy.Exists(temp):
                 arcpy.Delete_management(temp)


 def apply_quality_gates(cache_df: pd.DataFrame, report_dir: Path) -> None:
     """
     Apply quality gates to geocoded results. CRITICAL: Halts execution if fail rate exceeds 5% threshold.
     """ print("\n[4/6] Applying quality gates")

     # Calculate failure statistics
     null_coords = cache_df[
         cache_df["X_Coord"].isnull() | cache_df["Y_Coord"].isnull()
     ]
     fail_rate = (len(null_coords) / len(cache_df)) * 100

     # Generate quality report
     report_path = report_dir / QUALITY_REPORT_FILENAME
     report_dir.mkdir(parents=True, exist_ok=True)

     report_lines = [
         "=" * 70,
         "GEOCODING QUALITY REPORT",
         "=" * 70,
         f"Generated: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}",
         "",
         "SUMMARY:",
         f"  Total unique addresses: {len(cache_df):,}",
         f"  Successfully geocoded: {len(cache_df) - len(null_coords):,}",
         f"  Failed to geocode: {len(null_coords):,}",
         f"  Fail rate: {fail_rate:.2f}%",
         "",
         f"QUALITY GATE: {'PASS' if fail_rate <= GEOCODE_FAIL_THRESHOLD else 'FAIL'}",
         f"  Threshold: {GEOCODE_FAIL_THRESHOLD}%",
         f"  Result: {fail_rate:.2f}%",
         "",
     ]

     if len(null_coords) > 0:
         report_lines.append("FAILED ADDRESSES (Sample):")
         for idx, row in null_coords.head(20).iterrows():
             report_lines.append(f"  - {row['FullAddress']} (Status: {row.get('GeocodeStatus', 'U')})")
         if len(null_coords) > 20:
             report_lines.append(f"  ... and {len(null_coords) - 20} more")
         report_lines.append("")

         # Export failed addresses to CSV for inspection
         failed_csv_path = report_dir / FAILED_ADDRESSES_FILENAME
         null_coords.to_csv(failed_csv_path, index=False)
         report_lines.append(f"Full failed addresses list: {failed_csv_path}")

     report_lines.append("=" * 70)

     # Write report
     report_text = "\n".join(report_lines)
     report_path.write_text(report_text)

     # Print to console
     print(report_text)

     # CRITICAL: Halt if fail rate exceeds threshold
     if fail_rate > GEOCODE_FAIL_THRESHOLD:
         print("\n[ERROR] Geocoding fail rate exceeds 5% threshold!") print("        Action required: Inspect failed addresses before proceeding.") print(f"        Failed addresses CSV: {report_dir / FAILED_ADDRESSES_FILENAME}")
         sys.exit(1)

     print("\n      [PASS] Quality gate passed - proceeding to merge")


 def merge_coordinates_to_baseline(
     baseline_df: pd.DataFrame,
     cache_df: pd.DataFrame
 ) -> pd.DataFrame:
     """Merge geocoded X/Y coordinates back to full baseline dataset.""" print("\n[5/6] Merging coordinates to baseline dataset")

     # Merge on FullAddress
     final_df = baseline_df.merge(
         cache_df[["FullAddress", "X_Coord", "Y_Coord"]],
         on="FullAddress",
         how="left"
     )

     # Verify merge
     records_with_coords = final_df[
         final_df["X_Coord"].notna() & final_df["Y_Coord"].notna()
     ]

     print(f"      Total records: {len(final_df):,}")
     print(f"      Records with coordinates: {len(records_with_coords):,}")
     print(f"      Coverage: {(len(records_with_coords) / len(final_df)) * 100:.2f}%")

     return final_df


 def export_cached_baseline(final_df: pd.DataFrame, output_path: Path) -> None:
     """Export geocoded baseline to Excel with SHA256 hash generation.""" print(f"\n[6/6] Exporting geocoded baseline")
     print(f"      Output: {output_path}")

     # Create output directory
     output_path.parent.mkdir(parents=True, exist_ok=True)

     # Export to Excel (write_only mode for memory efficiency)
     start_time = time.time()
     final_df.to_excel(
         output_path,
         sheet_name="Sheet1",
         index=False,
         engine="openpyxl"
     )

     elapsed = time.time() - start_time
     file_size_mb = output_path.stat().st_size / (1024 * 1024)

     print(f"      Exported {len(final_df):,} records in {elapsed:.1f} seconds")
     print(f"      File size: {file_size_mb:.1f} MB")

     # Calculate SHA256 hash
     print("\n      Calculating SHA256 hash for integrity verification...")
     sha256_hash = calculate_sha256(output_path)
     print(f"      SHA256: {sha256_hash}")

     # Write hash to companion file
     hash_file = output_path.with_suffix(".xlsx.sha256")
     hash_file.write_text(f"{sha256_hash}  {output_path.name}\n")
     print(f"      Hash saved to: {hash_file.name}")
