```python
# Script: quick_consolidate_2019_2026.py
import pandas as pd
from pathlib import Path

# Source files from 05_EXPORTS\_CAD
yearly_files = [
    "yearly/2019/2019_CAD_ALL.xlsx",
    "yearly/2020/2020_CAD_ALL.xlsx",
    "yearly/2021/2021_CAD_ALL.xlsx",
    "yearly/2022/2022_CAD_ALL.xlsx",
    "yearly/2023/2023_CAD_ALL.xlsx",
    "yearly/2024/2024_CAD_ALL.xlsx",
    "yearly/2025/2025_CAD_ALL.xlsx",
    "monthly/2026/2026_01_01_to_2026_01_29_CAD_Export.xlsx"
]

# Load and concatenate
dfs = []
for file in yearly_files:
    df = pd.read_excel(CAD_ROOT / file)
    dfs.append(df)

consolidated = pd.concat(dfs, ignore_index=True)

# Filter date range: 2019-01-01 to 2026-01-29
consolidated['TimeOfCall'] = pd.to_datetime(consolidated['TimeOfCall'])
consolidated = consolidated[
    (consolidated['TimeOfCall'] >= '2019-01-01') &
    (consolidated['TimeOfCall'] <= '2026-01-29')
]

# Export
consolidated.to_csv(
    "CAD_Data_Cleaning_Engine/data/01_raw/2019_to_2026_01_29_CAD.csv",
    index=False
)

print(f"✅ Consolidated {len(consolidated):,} records")
print(f"   Date range: {consolidated['TimeOfCall'].min()} to {consolidated['TimeOfCall'].max()}")
```
